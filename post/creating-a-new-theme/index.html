尾牙重複抽獎期望計算

40人抽100獎項可重複抽取

某人都沒中的機率

(39/40)^100 ~= 7.951728986%

40個人裡有人沒中的機率

數學精算

$C_1^{40}(39/40)^{100}-C_2^{40}(38/40)^{100}+C_3^{40}(37/40)^{100}…C_{39}^{40}(1/40)^{100} ~= 0.9768650966830548$

simulator ~= 97.3%

package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {
	rand.Seed(time.Now().UnixNano())
	count := 0
	times := 100000
	for i := 0; i < times; i++ {
		if getDev() {
			count++
		}
	}
	fmt.Println(float64(count) / float64(times))
}

func getDev() bool {
	stat := map[int]int{}
	for i := 0; i <= 100; i++ {
		a := rand.Intn(40)
		stat[a]++
	}

	for i := 0; i < 40; i++ {
		if stat[i] == 0 {
			return true
		}
	}
	return false
}
